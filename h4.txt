CREATE TABLE products (
                          id SERIAL PRIMARY KEY,
                          productName VARCHAR(50) NOT NULL,
                          manufacturer VARCHAR(50) NOT NULL,
                          productCount INTEGER DEFAULT 0,
                          price NUMERIC,
                              isDiscounted BOOL
);
